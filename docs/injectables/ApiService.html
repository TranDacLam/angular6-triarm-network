import {Injectable} from '@angular/core';
import {HttpClient, HttpHeaders} from '@angular/common/http';
import {environment} from '../../../environments/environment';
import {JSONResponse} from '../types/json-response';
import {LogService} from './log.service';
import {ErrorHandler} from '../common/handle-error';

@Injectable({
  providedIn: 'root'
})
export class ApiService {

  static USER_TOKEN_KEY = 'USER_TOKEN';
  userToken: any = '';

  constructor(private http: HttpClient,
              private logger: LogService,
              private errorHandler: ErrorHandler,
  ) {
  }

  public async loadToken() {
    let tokenData = localStorage.getItem(ApiService.USER_TOKEN_KEY);
    if (tokenData) {
      tokenData = JSON.parse(tokenData);
      if (tokenData['tokenExpiration'] < Date.now()) {
        this.logger.log('refreshing token');
        await this.refreshToken({refreshToken: tokenData['refreshToken']});
      } else {
        this.userToken = tokenData;
      }
    } else {
      this.userToken = null;
    }
  }

  private async refreshToken({refreshToken}): Promise<JSONResponse> {
    const response = await
      this.postRequest({
        data: {
          action: 'get_oauth_token',
          grantType: 'refresh',
          grantData: [refreshToken]
        },
        options: {
          headers: new HttpHeaders({
            'Content-Type': 'application/json',
          })
        },
        noRetry: true
      });
    if (response.success) {
      this.logger.log('updated new token', response.result);
      // -- save user
      this.userToken = response.result;
      // -- save to localStorage
      localStorage.setItem(ApiService.USER_TOKEN_KEY, JSON.stringify(this.userToken));
    } else {
      this.userToken = null;
    }
    return response;
  }

  private async postRequest({data, options, noRetry}): Promise<JSONResponse> {
    this.logger.log(data, options, noRetry);
    return new Promise<JSONResponse>((resolve) => {
      const newHeaders = options.headers;
      // if (this.userToken) {
      //   newHeaders = newHeaders.append('Authorization', this.userToken);
      // }
      this.http.post<any>(environment.API_SERVER + '/rpc', data, Object.assign({}, options, {
        headers: newHeaders,
        responseType: 'json'
      })).subscribe(
        async (response) => {
          this.logger.log(response);
          if (!response['success'] && !noRetry) {

            // if (['SYSTEM_ERROR'].indexOf(response['error']['code']) >= 0) {
            //     this.handleError.handle_error('SYSTEM_ERROR');
            //     return;
            // }

            if (!!this.userToken && ['ACCESS_DENIED', 'ACCESS_CONDITION_FAILED'].indexOf(response['error']['code']) >= 0) {

              // if (['ACCESS_DENIED'].indexOf(response['error']['code']) >= 0) {
              //     this.handleError.handle_error('ACCESS_DENIED');
              //     return;
              // }

              // -- retry
              this.logger.log('refreshing token', this.userToken);
              const refreshResult = await this.refreshToken({refreshToken: this.userToken['refreshToken']});
              if (!refreshResult.success) {
                // if (['INVALID_TOKEN'].indexOf(refreshResult['error']['code']) >= 0) {
                //     this.removeToken();
                //     this.errorHandler.handle('INVALID_TOKEN');
                //     return;
                // }
                return resolve(new JSONResponse(refreshResult));
              }

              const result = await this.postRequest({data, options, noRetry: true});
              return resolve(new JSONResponse(result));
            }
          }
          console.log('response', response);
          return resolve(new JSONResponse({
            extra: response['extra'] ? response['extra'] : null,
            success: response['success'],
            result: response['result'] ? response['result'] : response['data'],
            error: response['error']
          }));
        },
        async (err) => {
          this.logger.log('err', err);
          return resolve(new JSONResponse({
            success: false,
            error: {errorCode: null, errorMessage: 'NETWORK_ERROR'}
          }));
        });
    });
  }

  async postJSON(data): Promise<JSONResponse> {
    if (this.userToken) {
      data.token = this.userToken;
    }
    return this.postRequest({
      data,
      options: {
        headers: new HttpHeaders({
          'Content-Type': 'application/json',
        })
      },
      noRetry: false
    });
  }


  async login(data_login): Promise<JSONResponse> {
    const data = {
      name: 'sign_in',
      params: data_login
    };
    const response = await this.postJSON(data);
    if (response.success) {
      // -- save user
      this.userToken = response.result['access_token'];
      localStorage.setItem(ApiService.USER_TOKEN_KEY, JSON.stringify(this.userToken));
    }
    return response;
  }

  async logout(): Promise<any> {
    this.userToken = null;
    localStorage.removeItem(ApiService.USER_TOKEN_KEY);
  }
}

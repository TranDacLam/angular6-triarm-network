                    $scope.close = function () {

                    };
                    
                    $scope.OK = function () {
                        var data = {
                            selectedAsset: $scope.selectedAsset,
                            selectedWallet: $scope.selectedWallet
                        }
                        $modalInstance.close(data);
                    };
                    $scope.changeWallet = function (selectedWallet) {
                        $scope.selectedAsset = null;
                        $scope.selectedWallet = selectedWallet;
                    };
                    $scope.selectAsset = function (asset) {
                        $scope.selectedAsset = asset;
                    };
                    
                    $scope.filterAsset = function (filter) {
                        filter = filter.toLowerCase();
                        $scope.filtering = true;
                        if ($scope.otherAssets) {
                            $scope.filteredAssets = $scope.otherAssets.filter(function (asset) {
                                var u1 = asset.name ? asset.name.toLowerCase().indexOf(filter) >= 0 : false;
                                var u2 = false; //asset.issuer && asset.issuer.name.toLowerCase().indexOf(filter);
                                var u3 = false; //asset.issuer && asset.issuer.address.toLowerCase().indexOf(filter);
                                return (filter === '@' || (filter && (u1 || u2 || u3)));
                            });
                        }
                        $scope.filtering = false;
                    };
                    
                    $scope.init = function () {
                        //-- request to get all user's asset and
                        var data = {
                            options: {},
                            commands: [
                                {
                                    name: 'get_all_assets',
                                    params: {}
                                },
                                {
                                    name: 'get_all_user_wallets',
                                    params: {
                                        with_assets: true
                                    }
                                },
                                {
                                    name: 'get_all_user_assets',
                                    params: {withoutAdvance: options.withoutAdvance}
                                }
                            ]
                        };
                    
                        $http.post('/batch', data).then(
                            function (response) {
                                $scope.loading = false;
                                console.log(response.data);
                                if (response.data.success) {
                                    if (response.data.results[0].error && response.data.results[0].error.errorCode === 12)
                                        document.location.href = '/login?redirect=' + encodeURIComponent(window.location.pathname);
                                    $scope.allAssets = response.data.results[0].success ? response.data.results[0].result : [];
                                    $scope.wallets = response.data.results[1].success ? response.data.results[1].result : [];
                                    $scope.issuingAssets = response.data.results[2].success ? response.data.results[2].result : [];
                    
                                    //-- extract ownedAssets
                                    $scope.ownedAssets = [];
                                    $scope.wallets.some(function (wallet) {
                                        if (wallet.balances) {
                                            wallet.balances.forEach(function (balance) {
                                                //-- check if ownedAssets has already contained this asset
                                                var contain = false;
                                                for (b in $scope.ownedAssets) {
                                                    if ($scope.ownedAssets[b]._id === balance._id) {
                                                        contain = true;
                                                        break;
                                                    }
                                                }
                                                if (!contain) $scope.ownedAssets.push(balance);
                                            });
                                        }
                                    });
                    
                                    $scope.otherAssets = [];
                                    $scope.allAssets.forEach(function (asset) {
                                        var contain = $scope.ownedAssets.some(function (ownAsset) {
                                            if (ownAsset._id === asset._id) return true;
                                        });
                                        if (!contain) {
                                            $scope.otherAssets.push(asset);
                                        }
                                    });
                                }
                            },
                            function (err) {
                                $scope.loading = false;
                            }
                        );
                    };
                    
                    $scope.init();
                    }]);
